class Solution {
    private Map<Integer, List<int[]>> graph;
    private int[] distances;
    private boolean[] visited;

    public int[] min(int n, int[][] edges) {
        graph = new HashMap<>();
        distances = new int[n];
        visited = new boolean[n];
        Arrays.fill(distances, Integer.MAX_VALUE);
        distances[0] = 0; // Distance to the start node is 0

        // æ„å»ºæ— å‘å›¾çš„é‚»æ¥è¡¨è¡¨ç¤ºï¼Œæ¯æ¡è¾¹æ·»åŠ ä¸¤æ¬¡
        for (int[] edge : edges) {
            int src = edge[0], dest = edge[1], weight = edge[2];
            graph.computeIfAbsent(src, k -> new ArrayList<>()).add(new int[]{dest, weight});
            graph.computeIfAbsent(dest, k -> new ArrayList<>()).add(new int[]{src, weight});
        }
        shortestPathsFromSource();
        for (int i = 0; i < distances.length; i++) {
            if (distances[i] == Integer.MAX_VALUE) distances[i] = -1;
        }
        return distances;
    }

    public void shortestPathsFromSource() {
        PriorityQueue<int[]> pq = new PriorityQueue<>(Comparator.comparingInt(a -> a[1]));
        pq.offer(new int[]{0, 0}); // [èŠ‚ç‚¹ç¼–å·, åˆ°è¯¥èŠ‚ç‚¹çš„è·ç¦»]

        while (!pq.isEmpty()) {
            int[] current = pq.poll();
            int currentNode = current[0];

            if (visited[currentNode]) continue;

            visited[currentNode] = true;

            List<int[]> neighbors = graph.getOrDefault(currentNode, Collections.emptyList());
            for (int[] neighbor : neighbors) {
                int nextNode = neighbor[0];
                int edgeWeight = neighbor[1];

                if (!visited[nextNode]) {
                    int newDist = distances[currentNode] + edgeWeight;
                    if (newDist < distances[nextNode]) {
                        distances[nextNode] = newDist;
                        pq.offer(new int[]{nextNode, newDist});
                    }
                }
            }
        }
    }
}

    public Node cloneGraph(Node node) {
        if (node == null) return null;
      //å€ŸåŠ©é˜Ÿåˆ—æ¥å®ç°å¹¿åº¦ä¼˜å…ˆéå†
        Deque<Node> q = new LinkedList<>();
        q.offer(node);
       //é‚»æ¥è¡¨ ç”¨æ¥å­˜å‚¨é¡¶ç‚¹çš„é‚»å±…é¡¶ç‚¹
        Map<Integer, Node> map = new HashMap<>();
        map.put(node.val, new Node(node.val));
        while (!q.isEmpty()) {
           Node source = q.poll();
           List<Node> nodes = source.neighbors;
           Node dest = map.get(source.val);
           for (Node n: nodes) {
               if (!map.containsKey(n.val)) {
                   q.offer(n);
                   map.put(n.val, new Node(n.val));
               }
               dest.neighbors.add(map.get(n.val));
           } 
        }
        return map.get(node.val);
    }
