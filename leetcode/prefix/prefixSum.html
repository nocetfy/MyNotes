            class Solution {
    public int totalStrength(int[] strength) {
        int n = strength.length;
        long mod = (long) 1e9 + 7, res = 0;
        int[] left = new int[n], right = new int[n];
        Arrays.fill(right, n);
        Deque<Integer> stack = new ArrayDeque<>(List.of(-1));
        for (int i = 0; i < n; i++) {
            while (stack.size() > 1 && strength[i] <= strength[stack.peek()]) {
                right[stack.pop()] = i;
            }
            left[i] = stack.peek();
            stack.push(i);
        }
        long sum = 0;
        long[] prefix = new long[n + 2];
        for (int i = 1; i <= n; i++) {
            sum += strength[i - 1];
            prefix[i + 1] = (prefix[i] + sum) % mod;
        }

        for (int i = 0; i < n; i++) {
            int l = left[i] + 1;
            int r = right[i] - 1;
            long total = ((i - l + 1) * (prefix[r + 2] - prefix[i + 1]) - (r - i + 1) * (prefix[i + 1] - prefix[l])) % mod;
            res = (res + total * strength[i]) % mod;
        }
        return (int) ((res + mod) % mod);
    }
}
