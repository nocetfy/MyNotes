    public boolean checkArray(int[] nums, int k) {
        int n = nums.length, sumDiff = 0;
        int[] diff = new int[n + 1];
        for (int i = 0; i < nums.length; i++) {
            sumDiff += diff[i];
            int x = nums[i] + sumDiff;
            if (x == 0) {
                continue;
            } else if (x < 0) {
                return false;
            } else {
                if (i + k > n) {
                    return false;
                }
                sumDiff -= x;
                diff[i + k] += x;
            }
        }
        return true;
    }
