              class Solution {
    public int[] rearrangeBarcodes(int[] barcodes) {
        PriorityQueue<Map.Entry<Integer, Integer>> queue =
                new PriorityQueue<>(Map.Entry.<Integer, Integer>comparingByValue().reversed());
        Map<Integer, Integer> map = new HashMap<>();
        for (int barcode : barcodes) {
            map.merge(barcode, 1, Integer::sum);
        }
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            queue.offer(entry);
        }
        int[] res = new int[barcodes.length];
        for (int i = 0; !queue.isEmpty();) {
            Map.Entry<Integer, Integer> poll = queue.poll();
            if (queue.isEmpty()) {
                if (poll.getValue() == 1) {
                    res[i++] = poll.getKey();
                }
            } else {
                Map.Entry<Integer, Integer> next = queue.poll();
                res[i++] = poll.getKey();
                res[i++] = next.getKey();
                if (poll.getValue() > 1) {
                    queue.offer(Map.entry(poll.getKey(), poll.getValue() - 1));
                }
                if (next.getValue() > 1) {
                    queue.offer(Map.entry(next.getKey(), next.getValue() - 1));
                }
            }
        }
        return res;

              class MedianFinder {
    PriorityQueue<Integer> min = new PriorityQueue<>(), max = new PriorityQueue<>(Comparator.reverseOrder());

    public void addNum(int num) {
        max.offer(num);
        min.offer(max.poll());
        if (max.size() < min.size()) {
            max.offer(min.poll());
        }
    }

            return max.peek();
        }
        return (double) (min.peek() + max.peek()) / 2;

    public int[] kthSmallestPrimeFraction(int[] arr, int k) {
        int n = arr.length;
        PriorityQueue<int[]> heap = new PriorityQueue<>((a, b) -> {
            double x = arr[a[0]] * 1.0 / arr[a[1]], y = arr[b[0]] * 1.0 / arr[b[1]];
            return Double.compare(x, y);

            int i = poll[0], j = poll[1];
            if (i + 1 < j) {
                heap.add(new int[]{i + 1, j});
            }
        }
        return new int[]{arr[heap.peek()[0]], arr[heap.peek()[1]]};
