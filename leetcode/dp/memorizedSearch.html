    Map<Integer, Integer> map = new HashMap<>();

    public int minOperations(int n) {
        if (map.containsKey(n)) return map.get(n);
        if ((n & (n - 1)) == 0) {
            map.put(n, 1);
            return 1;
        }
        int lowbit = n & -n;
        return Math.min(minOperations(n + lowbit), minOperations(n - lowbit)) + 1;
    }
