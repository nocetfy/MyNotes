    public int beautifulNumbers(int l, int r) {
        String lo = Integer.toString(l);

        int n = hi.length();
        lo = "0".repeat(n - lo.length()) + lo;

    int dfs(int i, int sum, int mul, boolean limitLow, boolean limitHigh, boolean isNum, char[] lo, char[] hi) {

        if (!limitHigh && !limitLow && isNum) {
            memo.put(mask, res);
        }

        return res;
    }
