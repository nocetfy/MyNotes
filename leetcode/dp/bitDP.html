                                          # 数位DP

                                          [toc]

                                          # 思路

                                          1. 数位概念：是指把一个数字按照个、十、百、千一位一位拆开，关注它每一位上的数字。

                                            举个例子： 如果拆的是十进制数，那么每一位数字都是 0-9， 其他进制可类比十进制。例如二进制，每一位数字都是0-1。

                                            leetcode 目前题目都是基于十进制和二进制的。

                                          2. 数位 DP：（题目应用特征和条件） （1）计数

                                            （2）输入会提供一个 数字区间(或者只提供上界) 作为计数的限制

                                            （3）题目条件经过 转换 后可以使用「数位」的思想去判断和理解

                                          3. 为什么可以用数位DP

                                            （1）最朴素的计数就是从小到大开始依次加一。

                                            （2）但我们发现对于 位数比较多的数，这样的过程中有许多重复的部分 。

                                            举个例子：从 700 数到 799、从 800 数到 899、和从 900 数到 999 的过程非常相似，它们都是后两位从 000 变到 999，不一样的地方只有百位这一位。所

                                            我们可以把重复计算的过程归并保证不重复计算，即将这些过程中产生的计数答案存放在一个数组里。此数组根据题目具体要求设置状态，用递推 或 DP 的

                                            式进行状态转移。

                                            （3）统计答案建议使用 记忆化搜索 ，为了 统计所有不超过上限的答案 ，要 从高到低枚举每一位 ，再考虑每一位都可以填哪些数字，最后利用通用答案数

                                            统计答案。

                                          数位dp的尝试方法其实并不特殊，绝大多数都是线性展开，尝试从左往右的尝试。解答的核心在于：可能性的整理、排列组合的相关知识。

                                          # 方法

                                          • 相对固定参数设置
                                          • 假设 n 是范围的最大值
                                          • i（必选） 表示第 i 个数位
                                          • isLimit （必选） 表示当前是否受到了n 的约束。若为真，则第i 位填入的数字至多为 s[i] ，否则至多为 9 。举个例子： n=234 ，如果前面填了 23 ，那么最后一位至多填 4 ；如果前面填的不是 23 ，那么最后一位至多填 9。如果在受到约束的情况下填了 s[i] ，那么后续填入的数字仍会受到n的约束。
                                          • isNum （可选） 表示i前面的数位是否填了数字。若为假，则当前位可以跳过（不填数字），或者要填入的数字至少为1 ；若为真，则要填入的数字可以从0开始。举个例子：例如 n=123 ，在i=0时跳过的话，相当于后面要构造的是一个99以内的数字了，如果i=1不跳过，那么相当于构造一个10 到99的两位数，如果i=1跳过，相当于构造的是一个9以内的数字。
                                          • mask（可选） 表示前面选过的数字集合，换句话说，第i位要选的数字不能在 mask 中。

                                          # 2376. 统计特殊整数open in new window

                                          class Solution {
    public int countSpecialNumbers(int n) {
        char[] s = Integer.toString(n).toCharArray();
        int[][] memo = new int[s.length][1 << 10];
        for (int[] row : memo) {
            Arrays.fill(row, -1); // -1 表示没有计算过
        }
        return dfs(0, 0, true, false, s, memo);
    }

    int dfs(int i, int mask, boolean isLimit, boolean isNum, char[] s, int[][] memo) {
        if (i == s.length) {
            return isNum ? 1 : 0; // isNum 为 true 表示得到了一个合法数字
        }
        if (!isLimit && isNum && memo[i][mask] != -1) {
            return memo[i][mask]; // 之前计算过
        }
        int res = 0;
        if (!isNum) { // 可以跳过当前数位
            res = dfs(i + 1, mask, false, false, s, memo);
        }
        // 如果前面填的数字都和 n 的一样，那么这一位至多填数字 s[i]（否则就超过 n 啦）
        int up = isLimit ? s[i] - '0' : 9;
        // 枚举要填入的数字 d
        // 如果前面没有填数字，则必须从 1 开始（因为不能有前导零）
        for (int d = isNum ? 0 : 1; d <= up; d++) {
            if ((mask >> d & 1) == 0) { // d 不在 mask 中，说明之前没有填过 d
                res += dfs(i + 1, mask | (1 << d), isLimit && d == up, true, s, memo);
            }
        }
        if (!isLimit && isNum) {
            memo[i][mask] = res; // 记忆化
        }
        return res;
    }
}

class Solution {
    HashMap<Long, Integer> memo = new HashMap<>();

    public int beautifulNumbers(int l, int r) {
        String lo = Integer.toString(l);
        String hi = Integer.toString(r);
        int n = hi.length();
        lo = "0".repeat(n - lo.length()) + lo;

        return dfs(0, 0, 1, true, true, false, lo.toCharArray(), hi.toCharArray());
    }

    int dfs(int i, int sum, int mul, boolean limitLow, boolean limitHigh, boolean isNum, char[] lo, char[] hi) {
        if (i == hi.length) {
            return mul % sum == 0 ? 1 : 0;
        }

        // i 8 3位 / sum 81 7位 / mul?
        // long mask = (long) mul << 32 | i << 16 | sum;
        long mask = (long) mul << 11 | (long) sum << 4 | i;
        if (!limitHigh && !limitLow && isNum && memo.containsKey(mask)) {
            return memo.get(mask);
        }

        int res = 0;
        if (!isNum && lo[i] == '0') {
            res += dfs(i + 1, sum, mul, true, false, false, lo, hi);
        } else {
            isNum = true;
        }

        int low = limitLow ? lo[i] - '0' : 0;
        int high = limitHigh ? hi[i] - '0' : 9;

        for (int d = isNum ? low : 1; d <= high; d++) {
            res += dfs(i + 1, sum + d, mul * d, limitLow && d == low, limitHigh && d == high, true, lo, hi);
        }

        if (!limitHigh && !limitLow && isNum) {
            memo.put(mask, res);
        }

        return res;
    }
}

                                          // 不补前导0 + 没有isNum参数
class Solution {
    Map<Long, Integer> memo = new HashMap<>();

    public int beautifulNumbers(int l, int r) {
        char[] low = String.valueOf(l).toCharArray(); // 无需补前导零
        char[] high = String.valueOf(r).toCharArray();
        return dfs(0, 1, 0, true, true, low, high);
    }

    int dfs(int i, int m, int s, boolean limitLow, boolean limitHigh, char[] low, char[] high) {
        if (i == high.length) {
            return s > 0 && m % s == 0 ? 1 : 0;
        }
        long mask = (long) m << 32 | (long) i << 16 | s; // 三个 int 压缩成一个 long
        if (!limitLow && !limitHigh && memo.containsKey(mask)) {
            return memo.get(mask);
        }

        int diff = high.length - low.length;
        int lo = limitLow && i >= diff ? low[i - diff] - '0' : 0;
        int hi = limitHigh ? high[i] - '0' : 9;

        int res = 0;
        int d = lo;
        if (limitLow && i < diff) { // 利用 limitLow 和 i 可以判断出 isNum 是否为 true，所以 isNum 可以省略
            res = dfs(i + 1, 1, 0, true, false, low, high); // 什么也不填
            d = 1; // 下面循环从 1 开始
        }
        for (; d <= hi; d++) {
            res += dfs(i + 1, m * d, s + d, limitLow && d == lo, limitHigh && d == hi, low, high);
        }

        if (!limitLow && !limitHigh) {
            memo.put(mask, res);
        }
        return res;
    }
}

                                          # 面试题 17.06. 2出现的次数open in new window

                                          编写一个方法，计算从 0 到 n (含 n) 中数字 2 出现的次数。

                                          示例:

                                          输入: 25
输出: 9
解释: (2, 12, 20, 21, 22, 23, 24, 25)(注意 22 应该算作两次)

                                          提示：

                                          • n <= 10^9

                                          思路：

                                          ​ 使用数位DP模板

                                          class Solution {
    char[] s;
    int[][] cache;

    public int numberOf2sInRange(int n) {
        s = Integer.toString(n).toCharArray();
        cache = new int[s.length][s.length];
        for (int[] array : cache) {
            Arrays.fill(array, -1);
        }
        return dfs(0, 0, true);
    }

    int dfs(int i, int count, boolean isLimit) {
        if (i == s.length) {
            return count;
        }
        if (!isLimit && cache[i][count] > -1) {
            return cache[i][count];
        }
        int up = isLimit ? s[i] - '0' : 9, res = 0;
        for (int d = 0; d <= up; d++) {
            res += dfs(i + 1, count + (d == 2 ? 1 : 0), isLimit && d == up);
        }
        if (!isLimit) {
            cache[i][count] = res;
        }
        return res;
    }
}
